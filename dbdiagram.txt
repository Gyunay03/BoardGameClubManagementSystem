Table Users {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
  password int
}

Table Events { 
  Number integer [primary key]
  Title varchar
  Type varchar
  TimeEvent timestamp
}


Table Reservation {
  Number integer [primary key]
  Name varchar
  Phone varchar
  Date timestamp
  Hour timestamp
  Gamers integer
  body text [note: 'Content of the post']
  Status varchar
  created_at timestamp
  Comment varchar
}

Table Menu {
  NumberOrder integer [primary key]
  Food varchar
  Drink varchar
  Cocktaile varchar
  Price currency
  created_at timestamp
  Comment varchar
}

Table BoardGames {
  Id integer [primary key]
  Name varchar
  Type varchar
  Gamers int
  Availability varchar
}

Table Orders {
  Id integer [primary key]
  NumberOrder integer
  ClientId integer
  GameId integer
  EventId integer
  created_at timestamp
}

Table Posetiteli {
  Id integer [primary key]
  Name varchar
  Phone varchar
  DateTime timestamp
  ReservationId integer
}


Ref: "BoardGames"."Id" < "Orders"."GameId"

Ref: "Events"."Number" < "Orders"."EventId"

Ref: "Posetiteli"."Id" < "Orders"."ClientId"

Ref: "Posetiteli"."ReservationId" < "Reservation"."Number"

Ref: "Menu"."NumberOrder" < "Orders"."NumberOrder"
