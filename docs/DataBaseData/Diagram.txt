Table Users {
  Id int [primary key]
  Username varchar
  Email varchar
  Role varchar
  Password varchar
  Created_at timestamp
}

Table Events { 
  EventNumber int [primary key]
  Title varchar
  Type varchar
  TimeEvent timestamp
  Created_at timestamp
}

Table Reservation {
  ReservationNumber int [primary key]
  UserId int
  Name varchar
  Phone varchar
  Date timestamp
  Hour timestamp
  GamersCount int
  Body text
  Status varchar
  Created_at timestamp
  Comment varchar
}

Table Menu {
  MenuId int [primary key]
  Category varchar
  ItemName varchar
  Price decimal
  Created_at timestamp
}

Table BoardGames {
  Id int [primary key]
  Name varchar
  Type varchar
  RequiredGamers int
  Availability varchar
}

Table Posetiteli {
  Id int [primary key]
  Name varchar
  Phone varchar
  ReservationId int
  DateAndTime timestamp
}

Table Orders {
  Id int [primary key]
  OrderNumber int
  ClientId int
  GameId int
  EventId int
  MenuId int
  Created_at timestamp
}

Ref: Orders.GameId > BoardGames.Id
Ref: Orders.EventId > Events.EventNumber
Ref: Orders.ClientId > Posetiteli.Id
Ref: Posetiteli.ReservationId > Reservation.ReservationNumber
Ref: Orders.MenuId > Menu.MenuId
Ref: Reservation.UserId > Users.Id
